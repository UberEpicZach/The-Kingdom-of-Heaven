#Triggers to control Coat of Arms weighting


coa_theocratic_trigger = {
	exists = root
	OR = {
		has_government = theocracy_government
		has_government = monastic_government
		AND = {
			exists = liege
			liege = {
				any_sub_realm_barony = {
					lessee = ROOT
					title_province = {
						has_holding_type = church_holding
					}
				}
			}
		}
	}
}
