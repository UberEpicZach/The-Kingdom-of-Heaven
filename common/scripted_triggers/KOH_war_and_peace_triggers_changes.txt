

can_be_warrior_trigger = {
	is_adult = yes
	is_imprisoned = no
	OR = {
	can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = $ARMY_OWNER$ }
		AND = {
			$ARMY_OWNER$.culture = {
				has_cultural_parameter = high_prowess_ignores_knight_restrictions
			}
			prowess >= 10
		}
	}
	NOT = {
		has_trait = incapable
		has_trait = KOH_Free_City #Added KOH
	}
}

can_be_knight_trigger = {
	trigger_if = {
		limit = {
			$ARMY_OWNER$.culture = {
				has_cultural_parameter = minimum_prowess_for_knights
			}
		}
		prowess >= 12
	}
	can_be_warrior_trigger = { ARMY_OWNER = $ARMY_OWNER$ }
	OR = {
		AND = {
			$ARMY_OWNER$.culture = {
				has_cultural_parameter = high_prowess_ignores_knight_restrictions
			}
			prowess >= 10
		}
	NOR = {
		AND = {
			has_trait = devoted
				OR = {
			faith = {
				NOT = {
					has_doctrine_parameter = clergy_can_fight
				}
			}
					culture = {
						NOT = {
							has_cultural_parameter = culture_clergy_can_fight
						}
					}
				}
		}
		AND = {
			is_clergy = yes
				OR = {
			faith = {
				NOT = {
					has_doctrine_parameter = clergy_can_fight
				}
			}
					culture = {
						NOT = {
							has_cultural_parameter = culture_clergy_can_fight
						}
					}
				}
		}
		has_trait = blind
		has_trait = KOH_Free_City #Added KOH
		}
	}
	is_ai = yes
} 

can_be_commander_basic_trigger = {
	is_alive = yes
	is_adult = yes
	NOR = {
		has_trait = incapable
		has_trait = KOH_Free_City #Added KOH
		AND = {
			has_trait = devoted
			faith = {
				NOT = {
					has_doctrine_parameter = clergy_can_fight
				}
			}
		}
		AND = {
			is_clergy = yes
			faith = {
				NOT = {
					has_doctrine_parameter = clergy_can_fight
				}
			}
		}
	}
	OR = { # You can always lead your own armies
		this = $ARMY_OWNER$
		can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = $ARMY_OWNER$ }
	}
	
	trigger_if = {
		limit = {
			$ARMY_OWNER$ = {
				is_ai = yes
			}
		}
		is_ai = yes
	}
}


is_religious_war = {
	OR = {
		using_cb = minor_religious_war
		using_cb = religious_war
		using_cb = major_religious_war
		using_cb = excommunication_war
		using_cb = undirected_great_holy_war
		using_cb = directed_great_holy_war
		using_cb = KOH_jihad #Added KOH
		using_cb = KOH_crusade #Added KOH
	}
}
