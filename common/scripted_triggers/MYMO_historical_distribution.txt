mymo_historic_bishopric = {
	OR = {
		AND = {
			NOT = { has_global_variable = arkomod_active }
			mymo_historic_bishopric_vanilla = yes
		}
		AND = {
			has_global_variable = arkomod_active
			mymo_historic_bishopric_arkomod = yes
		}
	}
}
