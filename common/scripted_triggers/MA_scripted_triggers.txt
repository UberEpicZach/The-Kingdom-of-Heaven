	#county is developed or in historically wealthy areas
	OR = {
		development_level >= 15
		title_province = {
			OR = {
				geographical_region = world_europe_south 				#Italy+Greece
				geographical_region = custom_ganges_plain 				#Ganges
				geographical_region = custom_bactria 					#Bactria

				geographical_region = world_middle_east_jerusalem 		#Levant

			}
		}
	}
}

required_innovation = {
	scope:holder.culture = { has_innovation = innovation_$INNOVATION$ }
}
MA_aesthetic_tradition = {
	custom_description = {
		text = MA_$TRADITION$_aesthetic
		exists = var:MA_$TRADITION$_culture
	}
}
required_religion = {
	scope:holder.religion = religion:$RELIGION$_religion
}
required_religious_family = {
	scope:holder.religion = { is_in_family = rf_$FAMILY$ }
}
