pool_assassin_order_succession = {
	valid_character = {
		is_capable_adult = yes
		is_imprisoned = no
		scope:base.faith = { has_allowed_gender_for_clergy = root }
		is_married = no
		is_betrothed = no
		NOT = { has_trait = excommunicated }
	}
	
	character_score = {
		base = 1
		
		modifier = {
			add = 1000
			is_courtier_of = scope:base
		}
		compare_modifier = {
			value = intrigue
		}
		modifier = {
			dynasty.dynasty_prestige_level >= 2
			add = {
				value = dynasty.dynasty_prestige_level
				multiply = 20
				subtract = 20
			}
		}
		modifier = {
			piety_level >= 2
			add = {
				value = piety_level
				multiply = 5
			}
		}
		modifier = {
			add = 100
			has_same_culture_as = scope:base
		}
		modifier = {
			add = 15
			culture = { has_same_culture_heritage = scope:base.culture }
		}
		modifier = {
			add = 10
			culture = { has_same_culture_language = scope:base.culture }
		}
		modifier = {
			add = 5
			culture = { has_same_culture_ethos = scope:base.culture }
		}
		modifier = {
			add = 50
			age > 35
		}
		modifier = {
			add = 100
			scope:base.faith = { has_preferred_gender_for_clergy = root }
		}
	}
	
	config = {
		background = intrigue
		age = { 25 45 }
	}
	
	selection_count = 25
	
	pool = faith
	gender = clergy
	court = yes
}
