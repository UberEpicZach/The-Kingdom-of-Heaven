KOH_guelph_ghibelline_map = {
	every_county = {
		if = {
			limit = {
				holder = {
					has_trait = KOH_ghibelline
				}
			}
			every_county_province = { barony = { set_color_from_title = title:d_tributary_map_ghibelline } }
		}
		else_if = {
			limit = {
				holder = {
					has_trait = KOH_guelph
				}
			}
			every_county_province = { barony = { set_color_from_title = title:d_tributary_map_gueplh } }
		}
		else = {
			every_county_province = { barony = { set_color_from_title = title:d_tributary_map_base } }
		}
	}
}
