MA_recycle_embellished_events = {
	if = {
		limit = {
			NOT = { exists = local_var:num_of_embellished_events }
		}
		set_local_variable = {
			name = num_of_embellished_events
			value = 1
		}
		trigger_event = {
			on_action = MA_embellish_city_events
			days = { 50 200 }
		}
	}
	else = {
		change_local_variable = {
			name = num_of_embellished_events
			add = 1
		}
		if = {
			limit = { # Check the number of events fired
				local_var:num_of_embellished_events < 2
			}
			trigger_event = {
				on_action = MA_embellish_city_events
				days = { 50 200 }
			}
		}
		else = {
            trigger_event = {
                id = MA_events_unique.0005
                days = { 50 200 }
            }
		}
	}
}
