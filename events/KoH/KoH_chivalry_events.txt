chivalry.1 = {
	type = character_event
	title = chivalry.1.t
	desc = chivalry.1.d
	theme = default
	left_portrait = {
		character = root
	}
	
	trigger = {
		is_landed = yes
	}

	option = {
		name = chivalry.1.a
		
		remove_gold = 100
		
		change_chivalry_effect = { 
			Chivalry = 1
		}
		custom_tooltip = add_1_chivalry_tt
		
		if = {
			limit = {
				has_trait = generous
			}
			add_stress = -10
		}
		else_if = {
			limit = {
				has_trait = greedy
			}
			add_stress = 10
		}
	}
	
	option = {
		name = chivalry.1.b
		
		add_gold = 10
		
		change_chivalry_effect = { 
			Chivalry = -1
		}
		custom_tooltip = remove_1_chivalry_tt
		
		add_tyranny = 5
		add_dread = 10
		
		if = {
			limit = {
				has_trait = generous
			}
			add_stress = 10
		}
		else_if = {
			limit = {
				has_trait = greedy
			}
			add_stress = -10
		}
	}
}

chivalry.2 = {
	type = character_event
	title = chivalry.2.t
	desc = chivalry.2.d
	theme = martial_chivalry_focus
	left_portrait = {
		character = root
	}
	
	right_portrait = {
		character = scope:challenger
	}

	trigger = {
		OR = {
			martial = 10
			AND = {
				exists = var:Chivalry
				var:Chivalry >= 3
			}
			
		}
	}

	immediate = {
		root = { random_dummy_gender_landholder_effect = yes }
        create_character = {
			age = { 20 30 }
			location = root.capital_province
			gender = scope:dummy_gender
			faith = this.faith
			culture = this.culture
			dynasty = none
			prowess = { 5 20 }
			save_scope_as = challenger
        }
	}
	
	option = {
		name = chivalry.2.a
		
		change_chivalry_effect = { 
			Chivalry = 1
		}
		custom_tooltip = add_1_chivalry_tt
		
		trigger_event = {
			id = chivalry.3
			days = 2
		}
	}
	
	option = {
		name = chivalry.2.b
		
		change_chivalry_effect = { 
			Chivalry = -1
		}
		custom_tooltip = remove_1_chivalry_tt
		
		add_prestige = -50
	}
}
