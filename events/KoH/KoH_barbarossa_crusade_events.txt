	type = character_event
	title = KoH_barbarossa_crusade.3.t

	theme = war
	override_background = {
		event_background = hills
	}
	left_portrait = {
		character = root
		animation = personality_honorable
	}
	right_portrait = {
		character = title:k_hungary.holder
		animation = personality_compassionate 
	}

	trigger = {
		NOT = {
			has_character_flag = has_had_KoH_barbarossa_crusade_3
		}
	}
	
	immediate = {
		add_character_flag = has_had_KoH_barbarossa_crusade_3
		hidden_effect = { every_army = { limit = { army_owner = root } set_army_location = province:3842 } }
	}

	option = {
		name = KoH_barbarossa_crusade.3.a
		trigger_event = {
			id = KoH_barbarossa_crusade.4
			days = 30
		}
		change_variable = {
			name = num_of_knights
			add = 2
		}
		custom_tooltip = slower_but_more_men

		if = {
			limit = {
				character:219517 = {
					AND = {
						is_alive = yes
						is_landed = yes
					}
				}
			}
			every_character_war = {
				limit = {
					primary_attacker = title:k_jerusalem.holder
					primary_defender = title:e_arabia.holder
				}
				add_attacker = character:219517
			}
		}
	}
	
	option = {
		name = KoH_barbarossa_crusade.3.b
		trigger_event = {
			id = KoH_barbarossa_crusade.4
			days = 15
		}
		custom_tooltip = faster_but_no_men
	}
}

KoH_barbarossa_crusade.11 = {
	type = character_event
	title = KoH_barbarossa_crusade.11.t
	desc = KoH_barbarossa_crusade.11.d
	
	theme = war
	override_background = {
		event_background = wilderness_wetlands
	}
	
	left_portrait = {
		character = root
		animation = shock
	}
	
	trigger = {
		NOT = {
			has_character_flag = has_had_KoH_barbarossa_crusade_11
		}
	}

	immediate = {
		add_character_flag = has_had_KoH_barbarossa_crusade_11
        every_character_war = {
            limit = {
                primary_defender = title:e_arabia.holder
                primary_attacker = title:k_jerusalem.holder
            }
            remove_participant = prev
        }
		hidden_effect = { every_army = { limit = { army_owner = root } set_army_location = root.capital_province } }
		title:k_jerusalem.holder = {
			every_character_war = {
				limit = {
					primary_attacker = title:k_jerusalem.holder
					primary_defender = title:e_arabia.holder
				}
				add_attacker = title:d_osterreich.holder
			}
		}
    }

	option = {
		name = KoH_barbarossa_crusade.11.a
		set_variable = {
			name = num_of_levies
			value = 5000
		}
		set_variable = {
			name = num_of_heavy_swords
			value = 0
		}
		set_variable = {
			name = num_of_crossbows
			value = 1
		}
		set_variable = {
			name = num_of_knights
			value = 1
		}
		title:d_osterreich.holder = {
			spawn_army = {
				uses_supply = no
				inheritable = yes
				name = barbarossa_crusaders
				men_at_arms = {
					type = jerusalem_knights
					stacks = var:num_of_knights
				}
				men_at_arms = {
					type = crossbowmen
					stacks = var:num_of_crossbows
				}
				men_at_arms = {
					type = armored_footmen
					stacks = var:num_of_heavy_swords
				}
				levies = {
					value = var:num_of_levies
				}
				location = province:5910
			}
		}
		death = {
			death_reason = death_drowned
		}
	}
}
