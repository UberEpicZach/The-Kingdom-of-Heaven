namespace = doublet
doublet.0001 = {
	title = doublet.0001.t
	desc = doublet.0001.desc
	theme = realm
	right_portrait = {
		character = scope:drunken_courtier
		animation = beg
	}

	immediate = {
		add_character_flag = {
			flag = had_event_doublet_trouble
			days = 1825
		}
		#Save the capital for loc. 
		capital_barony = { save_scope_as = capital_barony }
		#Create Drunk
		create_character = {
			location = root.capital_province
			template = robert_of_st_albans_character
			save_scope_as = drunken_courtier
			faith = faith:catholic
			culture = root.culture
			dynasty = no
			trait = drunkard
		}
		add_prestige = minor_prestige_loss
	}

	#be chill
	option = {
		name = doublet.0001.a
		add_prestige = medium_prestige_loss
		scope:drunken_courtier = {
			add_opinion = {
				target = root
				modifier = pleased_opinion
				opinion = 30
			}
		}
	}

	#be somewhat chill
	option = {
		name = doublet.0001.b
		add_prestige = minor_prestige_loss
		scope:drunken_courtier = {
			add_opinion = {
				target = root
				modifier = pleased_opinion
				opinion = 10
			}
		}
	}

	#shame them
	option = {
		name = doublet.0001.c
		add_prestige = minor_prestige_gain
		scope:drunken_courtier = {
			add_opinion = {
				target = root
				modifier = angry_opinion
				opinion = -10
			}
		}
	}

	#hurt them
	option = {
		name = doublet.0001.d
		add_prestige = medium_prestige_gain
		add_dread = minor_dread_gain
		scope:drunken_courtier = {
			add_opinion = {
				target = root
				modifier = angry_opinion
				opinion = -30
			}
		}
	}

	#kill them
	option = {
		name = doublet.0001.e
		add_prestige = medium_prestige_gain
		add_dread = 50
		add_prestige = medium_prestige_loss
		add_tyranny = 10
		execute_prisoner_effect = {
			VICTIM = scope:drunken_courtier
			EXECUTIONER = root
		}
	}
}
