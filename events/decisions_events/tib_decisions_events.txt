major_decisions_mosque_seville.1 = {
	type = character_event
	title = major_decisions_mosque_seville.1.t
	desc = {
		desc = major_decisions_mosque_seville.1.desc_first
		first_valid = {
			desc = major_decisions_mosque_seville.1.desc
		}
	}
	theme = stewardship
	left_portrait = {
		character = scope:founder
		animation = personality_rational
	}

	option = {
		trigger = {
			any_sub_realm_barony = { this = title:b_sevilla }
			title:b_sevilla = {
				county = { development_level >= 20 }
				title_province = { has_special_building_slot = no }
			}
		}
		title:b_sevilla.title_province = {
			title:b_sevilla = {
				if = {
					limit = { NOT = { holder = root } }
					holder = {
						add_opinion = {
							target = root
							modifier = pleased_opinion
							opinion = 30
						}
					}
				}
			}
		}
		name = major_decisions_mosque_seville.b_sevilla
	}

	option = { # fallback in case development somehow tumbles
		trigger = { always = no }
		fallback = yes
		name = major_decisions_mosque_seville.1.fallback
		custom_tooltip = major_decisions_mosque_seville.1.fallback.tt
	}
}
