CrusaderStateEvents.1 = { #
	title = CrusaderStateEvents.1.t
	desc = CrusaderStateEvents.1.desc
	theme = faith
	override_background = {
		event_background = army_camp
	}
	left_portrait = {
		character = this
		animation = marshal
	}

	immediate = {
		save_scope_as = war_attacker
		scope:war_attacker.primary_title = {
			save_scope_as = primary_title
		}
		faith = {
			great_holy_war = {
				ghw_target_title = { #Loc keys for localization fluff.
					save_scope_as = target_kingdom
				}
				random_pledged_attacker = {
					limit = {
						exists = ghw_beneficiary
						this = scope:war_attacker
					}
					ghw_beneficiary = {
						save_scope_as = beneficiary
					}
				}
				ghw_war = {
					primary_defender = {
						save_scope_as = defender
						random_sub_realm_county = {
							limit = {
								duchy = {
									any_in_de_jure_hierarchy = {
										count = all
										trigger_if = {
											limit = { tier = tier_county }
											holder.top_liege = scope:defender
											NOT  = { county_controller = holder }
											exists = var:occupant
											var:occupant = scope:war_attacker
										}
									}
								}
							}
							duchy = {
								save_scope_as = duchyforloc
							}
							save_scope_as = countyforloc
						}
					}
		
				}
			}
		}
	}

	option = {  #beneficiary 
		name = CrusaderStateEvents.1.a
		custom_tooltip = benefiticary_cruasder_state	
		WarDuchyEffect = yes
		scope:beneficiary = {
			add_character_modifier = {
				modifier = settledcrusadersgood
			   	years = 3
		   }
		}
		scope:war_attacker = {
			add_character_modifier = {
				modifier = settledcrusaders
			   	years = 3
		   }
		   add_character_flag = has_taken_crusader_state
		}

		if = {
			limit = {
				scope:duchyforloc.holder = {
					any_sub_realm_title = {
						title_province = {
							geographical_region = ghw_region_anatolia
							geographical_region = ghw_region_near_east
							geographical_region = ghw_region_mediterranean
							geographical_region = ghw_region_balkans
							geographical_region = ghw_region_holy_land
							geographical_region = ghw_region_iberia
							geographical_region = ghw_region_egypt
						}
					}
				}
			}
			scope:beneficiary = {
				change_government = CrusaderState
			}
		}
		# else = {
		# 	change_government = feudal_government 
		# }
			
		hidden_effect = {
			scope:war_attacker = {
				if = {
					limit = {
						exists = var:CrusaderState_variable
					}
					remove_variable = CrusaderState_variable
				}
			}
		}

		scope:war_attacker.faith.great_holy_war = {
			#Toast for attackers
			every_pledged_attacker = {
				limit = {
					NOT = { THIS = root }
				}
				trigger_event = {
					id = CrusaderStateEvents.2
					days = 2
				}
			}
		}

		scope:war_attacker = {
			trigger_event = {
				id = CrusaderStateEvents.4
				days = 2
			}
		}

		scope:war_attacker = {
			add_character_flag = {
				flag = ghw_beneficiary_serving
				days = 550
			}
		}

		scope:war_attacker = {
			trigger_event = {
				id = CrusaderStateEvents.5
				days = 10
			}
		}

		ai_chance = {
			base = 80
			modifier = {
				add = 100
				has_trait = honest
			}
			ai_value_modifier = { #make the average dude take this ai_rationality = 0.5 ai_zeal = 1
				ai_rationality = 0.5
				ai_zeal = 1
			}
		}
	}

	option = { #self
		name = CrusaderStateEvents.1.b
		custom_tooltip = self_serving_cruasder_state

		scope:war_attacker = {
			add_character_modifier = {
				modifier = settledcrusaders
			   	years = 3
		   }
		   add_character_flag = has_taken_crusader_state
		}
		faith = {
			great_holy_war = {
				every_pledged_attacker = {
					limit = {
						NOT = { 
							has_relation_lover = root 
							has_relation_best_friend = root
						}
					}
					add_opinion = {
						modifier = opinion_selfish_crusader
						target = root
					} 
				}
			}
		}

		scope:war_attacker.faith.great_holy_war = {
			#Toast for attackers
			every_pledged_attacker = {
				limit = {
					NOT = { THIS = root }
				}
				trigger_event = {
					id = CrusaderStateEvents.2
					days = 2
				}
			}
		}

		hidden_effect = {
			scope:war_attacker = {
				if = {
					limit = {
						exists = var:CrusaderState_variable
					}
					remove_variable = CrusaderState_variable
				}
			}
		}

		WarDuchyEffect_forSelf = yes
		
		scope:war_attacker = {
			add_character_flag = {
				flag = ghw_self_serving
				days = 550
			}
		}
		ai_chance = {
			base = 80
			modifier = {
				add = 100
				has_trait = greedy
			}
			modifier = {
				add = 100
				has_trait = deceitful
			}
			ai_value_modifier = { #make the average dude take this ai_rationality = 0.5 ai_zeal = 1
				ai_rationality = 0.5
				ai_zeal = 1
			}
		} #make Greedy, dishonest characters, take this
	}

	option = { #I'll personally go, abdicate
		name = CrusaderStateEvents.1.c 

		custom_tooltip = adbicate_cruasder_state			
		
		hidden_effect = {
			WarDuchyEffect_forSelfButDepose = yes

			trigger_event = {
				days = 1 
				id = CrusaderStateEvents.3 
			}
		}

		scope:war_attacker.faith.great_holy_war = {
			#Toast for attackers
			every_pledged_attacker = {
				limit = {
					NOT = { THIS = root }
				}
				trigger_event = {
					id = CrusaderStateEvents.2
					days = 2
				}
			}
		}

		scope:war_attacker = {
			trigger_event = {
				id = CrusaderStateEvents.4
				days = 2
			}
			add_character_flag = has_taken_crusader_state
		}

		scope:war_attacker = {
			add_character_flag = {
				flag = ghw_personal_abdicate_serving
				days = 550
			}
		}

		hidden_effect = {
			scope:war_attacker = {
				if = {
					limit = {
						exists = var:CrusaderState_variable
					}
					remove_variable = CrusaderState_variable
				}
			}
		}
		add_trait = crusader_state_trait

		ai_chance = {
			base = 80
			modifier = {
				add = 100
				has_trait = humble
			}
			modifier = {
				add = 100
				has_trait = honest
			}
			modifier = {
				add = 100
				has_trait = temperate
			}
			modifier = {
				add = 100
				has_trait = just
			}
			ai_value_modifier = { #make the average dude take this ai_rationality = 0.5 ai_zeal = 1
				ai_zeal = 10
			}
		}
	}

	option = { #don't care
		name = CrusaderStateEvents.1.d
		custom_tooltip = leaving_cruasder_state
	}

CrusaderStateEvents.4 = { #Spawn Army?
	hidden = yes

	immediate = {
		if = {
			limit = {
				scope:war_attacker = { 
					has_character_flag = ghw_beneficiary_serving 
				}
			}

			scope:beneficiary = {
				spawn_army = {
					levies = {
						value = 100
						scope:war_attacker.faith.great_holy_war = {
							every_pledged_attacker = {
								add = 100
							}
						}
					}
					men_at_arms = {
						type = teutonic_knights
						stacks = {
							value = 1
							scope:war_attacker.faith.great_holy_war = {
								every_pledged_attacker = {
									add = 0.2
								}
							}
						}
					}
					inheritable = no
					uses_supply = no
					location = scope:beneficiary.capital_province
					name = CrusaderKnights
				}
			}
		}
		else_if = {
			limit = {
				scope:war_attacker = { 
					has_character_flag = ghw_personal_abdicate_serving 
				}
			}
			scope:war_attacker = {
				spawn_army = {
					levies = {
						value = 100
						scope:war_attacker.faith.great_holy_war = {
							every_pledged_attacker = {
								add = 100
							}
						}
					}
					men_at_arms = {
						type = teutonic_knights
						stacks = {
							value = 1
							scope:war_attacker.faith.great_holy_war = {
								every_pledged_attacker = {
									add = 0.2
								}
							}
						}
					}
					inheritable = no
					uses_supply = no
					location = scope:beneficiary.capital_province
					name = CrusaderKnights	
				}
			}
		}	
	}
}
