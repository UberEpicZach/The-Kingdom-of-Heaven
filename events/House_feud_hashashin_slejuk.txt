namespace = house_feud_hashashin
house_feud_hashashins.0001 = {
	type = character_event
	title = house_feud_hashashins.0001.t
	desc = {
		desc = house_feud_hashashins.0001.desc
		triggered_desc = {
			trigger = { exists = scope:other_house_feud_story }
			desc = house_feud_hashashins.0001.target
		}
	}
	theme = rival_relation
	override_background = { event_background = bedchamber }
	left_portrait = {
		character = character:227012
		animation = boredom
		outfit_tags = { nightgown }
	}
	lower_right_portrait = character:144056

	trigger = {
		#DLC check
		has_bp1_dlc_trigger = yes
		#Standard checks
		is_available_adult = yes
		# Is valid feuder
		house_feud_house_head_trigger = yes
		# Someone to feud with
		any_relation = {
			type = rival
			exists = house
			this = this.house.house_head
			house_feud_attacker_trigger = { VICTIM_HEAD = character:227012 }
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			any_relation = {
				type = rival
				house_feud_attacker_trigger = { VICTIM_HEAD = character:227012 }
				has_relation_nemesis = character:227012
			}
		}
		modifier = {
			factor = 2
			any_relation = {
				type = rival
				house_feud_attacker_trigger = { VICTIM_HEAD = character:227012 }
				primary_title.tier = character:227012.primary_title.tier
			}
		}
	}

	immediate = {
		play_music_cue = "mx_cue_negative"
		random_relation = {
			type = rival
			limit = {
				house_feud_attacker_trigger = { VICTIM_HEAD = character:227012 }
			}
			weight = {
				base = 1
				modifier = {
					factor = 1000
					has_relation_nemesis = character:227012
				}
				modifier = {
					factor = 2
					primary_title.tier = character:227012.primary_title.tier
				}
			}
			house_feud_save_head_and_attacker_effect = yes
		}
		house_feud_rival_story_effect = { RIVAL = character:144056 }
	}

	option = { # Start Feud
		name = house_feud_hashashins.0001.a
		house_feud_start_or_reciprocate_tooltip_effect = { FEUD_TARGET = character:144056 }
		random_owned_story = {
			limit = { story_type = story_cycle_house_feud }
			set_variable = {
				name = house_feud_reason
				value = flag:head_rivalry
			}
			house_feud_save_victim_attacker_effect = yes
		}

		ai_will_do = {

			base = 100
		}

		# Stress effects in scripted_effect
		ai_chance = {
			base = 25
			ai_value_modifier = {
				ai_boldness = 1
				ai_vengefulness = 1
			}
		}
	}

	option = { # Remain rivals
		name = house_feud_hashashins.0001.b
		house_feud_forgiving_stress_effect = yes
		ai_chance = {
			base = 25
			ai_value_modifier = {
				ai_boldness = -0.5
				ai_vengefulness = 0.5
			}
		}
	}
}
