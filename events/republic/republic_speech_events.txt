republic_speech_event.1005 = { # A Rival's Insult
	type = character_event
	title = republic_speech_event.1005.t
	desc = republic_speech_event.1005.desc
	left_portrait = {
		character = root
		animation = fear
	}
	right_portrait = {
		character = scope:your_rival
		animation = anger
	}
	theme = diplomacy
	override_background = {
		event_background = docks
	}
	trigger = {
		any_relation = {
			type = rival
			is_house_head = yes
			house = {
				exists = var:titular_republic
				var:titular_republic = root.house.var:titular_republic
			}
		}
	}
	immediate = {
		if = {
			limit = { house = { exists = var:titular_republic } }
			house = { var:titular_republic = { save_scope_as = titular_republic } }
		}
		random_relation = {
			type = rival
			limit = {
				is_house_head = yes
				house = {
					exists = var:titular_republic
					var:titular_republic = root.house.var:titular_republic
				}
			}
			save_scope_as = your_rival
		}
	}

	# Defend me
	option = {
		name = republic_speech_event.1005.a
		add_prestige = 50
		if = {
			limit = {
				OR = {
					AND = {
						is_independent_ruler = yes
						republic_has_council_form = yes
					}
					AND = {
						is_independent_ruler = no
						liege_or_court_owner = {
							republic_has_council_form = yes
						}
					}
				}
			}
			custom_description = {
				text = effect_change_noble_appeasement_positive
				value = 10
				change_noble_appeasement_positive = { AMOUNT = 10 }
			}
			specific_post_appeasement= { CLASS = nobility }
		}
		add_opinion = {
			target = scope:your_rival
			modifier = rival_insult_opinion
		}
	}

	# Defend the people
	option = {
		name = republic_speech_event.1005.b
		add_piety = 20
		random_list = {
			90 = {}
			10 = {
				add_trait = wounded_1
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						is_independent_ruler = yes
						republic_has_council_form = yes
					}
					AND = {
						is_independent_ruler = no
						liege_or_court_owner = {
							republic_has_council_form = yes
						}
					}
				}
			}
			custom_description = {
				text = effect_change_popular_appeasement_positive
				value = 10
				change_popular_appeasement_positive = { AMOUNT = 10 }
			}
			specific_post_appeasement= { CLASS = popular }
		}
		add_opinion = {
			target = scope:your_rival
			modifier = rival_insult_opinion
		}
	}
}
