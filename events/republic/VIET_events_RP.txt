# Buying new clothes for a ball?
VIETrepublic.0001 = {
	type = character_event
	title = VIETrepublic.0001.t
	desc = VIETrepublic.0001.desc
	theme = family
	
	right_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		house = {
			exists = var:titular_republic
			has_variable = titular_republic
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = arrogant
		}
		modifier = {
			add = -0.2
			has_trait = humble
		}
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = ambitious
		}
		modifier = {
			add = -0.2
			has_trait = content
		}
	}
	
	option = { # Buy opulent styles
		name = VIETrepublic.0001.a
		trigger = {
			OR = {
				gold >= 40
				is_ai = no
			}
		}
		remove_short_term_gold = 40
		add_character_modifier = {
			modifier = VIET_RP_ball_clothes_opulent
			years = 8
		}	
		stress_impact = {
			humble = minor_stress_impact_gain
			temperate = minor_stress_impact_gain
		}		
	}	
	option = { # Simple clothes
		name = VIETrepublic.0001.b
		trigger = {
			OR = {
				gold >= 40
				is_ai = no
			}
		}
		remove_short_term_gold = 40
		add_character_modifier = {
			modifier = VIET_RP_ball_clothes_simple
			years = 8
		}	
		stress_impact = {
			arrogant = minor_stress_impact_gain
			ambitious = minor_stress_impact_gain
		}		
	}	
	option = { # Exotic clothes
		name = VIETrepublic.0001.c
		trigger = {
			OR = {
				gold >= 40
				is_ai = no
			}
		}
		remove_short_term_gold = 40
		add_character_modifier = {
			modifier = VIET_RP_ball_clothes_exotic
			years = 8
		}	
		stress_impact = {
			temperate = minor_stress_impact_gain
			arrogant = minor_stress_impact_gain
		}		
	}	
	option = { # Non
		name = VIETrepublic.0001.d	
		if = {
			limit = {
				faith = {
					OR = {
						trait_is_sin = arrogant
						trait_is_virtue = humble
						trait_is_virtue = temperate
					}
				}
			}
			add_piety = 25
		}	
		stress_impact = {
			arrogant = minor_stress_impact_gain
			ambitious = minor_stress_impact_gain
			gregarious = minor_stress_impact_gain
		}
	}
}

# Courtier complains about nouveau riche 
VIETrepublic.0007 = {
	type = character_event
	title = VIETrepublic.0007.t
	desc = VIETrepublic.0007.desc
	theme = stewardship_wealth_focus
	
	left_portrait = root
	right_portrait = {
		character = scope:RP_snob
		animation = disapproval
	}
	
	trigger = {
		any_courtier = {
			is_available_ai_adult = yes
			NOT = { has_trait = humble }
			NOT = { has_trait = compassionate }
			OR = {
				has_trait = arrogant
				has_trait = wrathful
				has_trait = stubborn
				has_trait = deceitful
				has_trait = callous
			}
			is_lowborn = no
			NOT = { is_close_family_of = root }
		}
		house = {
			exists = var:titular_republic
			has_variable = titular_republic
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = diligent
		}
		modifier = {
			add = -0.2
			has_trait = lazy
		}
		modifier = {
			add = 0.2
			has_trait = patient
		}
		modifier = {
			add = -0.2
			has_trait = impatient
		}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				is_available_ai_adult = yes
				NOT = { has_trait = humble }
				NOT = { has_trait = compassionate }
				OR = {
					has_trait = arrogant
					has_trait = wrathful
					has_trait = stubborn
					has_trait = deceitful
					has_trait = callous
				}
				is_lowborn = no
				NOT = { is_close_family_of = root }
			}
			save_scope_as = RP_snob
			add_character_modifier = {
				modifier = VIET_RP_noveau_riche_partner
				years = 5
			}
		}
	}
	
	option = { # At least he helps you make money, right?
		name = VIETrepublic.0007.a
		stress_impact = {
			arrogant = minor_stress_impact_gain
		}
	}	
	option = { # I, too, am a snob
		name = VIETrepublic.0007.b
		trigger = {
			NOT = { has_trait = humble }
		}
		add_prestige = 25
		add_piety = -25
		stress_impact = {
			compassionate = minor_stress_impact_gain
		}
	}	
	option = { # Let us not be snobs
		name = VIETrepublic.0007.c
		trigger = {
			NOT = { has_trait = arrogant }
		}
		add_piety = 25
		add_prestige = -25
		stress_impact = {
			callous = minor_stress_impact_gain
		}
	}
}
